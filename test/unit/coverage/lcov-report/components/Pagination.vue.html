<template>
  <div class="pagination" v-if="count !== 0">
    <ul>
      <li
        @click="prev()"
        v-if="arrowNumber !== 0"
      >
        <a><v-icon>keyboard_arrow_left</v-icon></a>
      </li>
 
      <li
        v-for="k in pagArray"
        :class="{'active': (arrowNumber * 10) + k === pagNumber}"
        @click="pageChange((arrowNumber * 10) + k)"
      >
        <a>{{ (arrowNumber * 10) + k }}</a>
      </li>
 
      <li
        @click="next()"
        v-if="arrowNumber !== arrowMax"
      >
        <a><v-icon>keyboard_arrow_right</v-icon></a>
      </li>
    </ul>
  </div>
</template>
 
<script>
let pag,
    pagSplit;
export default {
  name: 'Page',
  data () {
    return {
      pagNumber: 1,
      pagArray: 0,
      arrowMax: 0,
      arrow: 0,
      arrowNumber: 0
    }
  },
  props: [
    'count',
    'amount'
  ],
  watch: {
    count () {
      this.setPag();
    },
    amount () {
      this.pagNumber = 1;
      this.arrowNumber = 0;
      this.setPag();
    }
  },
  methods: {
    setPag () {
      pag = Math.ceil(this.count/ this.amount),
      pagSplit = (pag/10).toString().split('.');
      this.pagArray = pag > 10 ? 10 : pag;
      this.arrowMax = Math.ceil(pag/10) - 1;
    },
    next () {
      this.arrowNumber = this.arrowMax === this.arrowNumber ? this.arrowNumber : this.arrowNumber+=1;
      this.pagArray = this.arrowMax === this.arrowNumber ? Number(pagSplit[1]) : 10;
      this.pageChange((this.arrowNumber * 10) + 1);
    },
    prev () {
      this.arrowNumber = 0 === this.arrowNumber ? 0 : this.arrowNumber-=1;
      this.pagArray = 10;
      this.pageChange((this.arrowNumber * 10) + 10);
    },
    pageChange (num) {
      this.$EventBus.$emit('changePage', num);
      this.pagNumber = num;
    }
  }
};
</script>
