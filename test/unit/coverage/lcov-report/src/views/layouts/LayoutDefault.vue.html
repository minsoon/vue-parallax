<template>
  <div id="wrap">
    <!-- @slot Use this slot header -->
    <app-header />
    <!-- @slot Use this slot content -->
    <v-content>
      <div class="titBox">
        <div class="box">
          <h1 class="title" :class="{show: showTitle}"><p><span>{{ oldTitle }}</span><span>{{ newTitle }}</span></p></h1>
          <div class="pathBox">
            <span>HOME</span> <span>{{ dep_one }}</span> <span v-if="dep_tow">{{ dep_tow }}</span>
          </div>
        </div>
      </div>
      <v-container fill-height>
        <transition name="fade-transition" mode="out-in">
          <router-view />
        </transition>
      </v-container>
    </v-content>
    <!-- @slot Use this slot footer -->
    <app-footer />
  </div>
</template>
 
<script>
import AppHeader from '@/components/AppHeader';
import AppFooter from '@/components/AppFooter';
 
export default {
  name: 'LayoutDefault',
  components: {
    AppHeader,
    AppFooter
  },
  data () {
    return {
      showTitle: false,
      newTitle: '',
      oldTitle: '',
      dep_one: this.$router.history.current.matched[0].meta.title,
      dep_tow: this.$router.history.current.matched[1] ? this.$router.history.current.matched[1].meta.title : false
    }
  },
  watch: {
    '$route' (to) {
      this.dep_one = to.matched[0].meta.title
      this.dep_tow = to.matched[1] ? to.matched[1].meta.title : false;
    }
  },
  created () {
    this.$EventBus.$on('app-title', (val) => {
      this.newTitle = val;
      this.showTitle = true;
      setTimeout(() => {
        this.oldTitle = val;
        this.showTitle = false;
      }, 300);
    });
  },
};
</script>
