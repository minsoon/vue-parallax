import { Doughnut } from 'vue-chartjs';
 
export default {
  extends: Doughnut,
  props: ['doughnutData'],
  watch: {
    doughnutData (val) {
      this.setGraph(val);
    }
  },
  methods: {
    setGraph (data) {
      this.renderChart({
        labels: ['유료', '무료', '해지'],
        datasets: [{
          backgroundColor: ['#E46651', '#00D8FF', '#DD1B16'],
          data: [data, 80, 10]
        }]
      }, {
        responsive: true,
        maintainAspectRatio: false
      });
    }
  }
};
