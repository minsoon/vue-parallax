  rank: false
};
 
export default {
  components: {
    ...VCheckbox,

  },
  data () {
    return {
      mode: false,
      onoff: false,
      userId: '',
      userList: false,
      userData: {
        name: '',
        id: '',
        rank: false
      }
    };
  },
  props: ['popupOpen'],
  watch: {
    popupOpen (val) {
      if (val !== false) {
        this.mode = val === 'new' ? false : true;
        this.userData = val === 'new' ? dataSet : val;
        this.onoff = true;
        this.$parent.popupOpen = false;
      }
    }
  },
  created () {
  },
  methods: {
    close () {
