import * as VNavigationDrawer from 'vuetify/es5/components/VNavigationDrawer';
 
export default {
  name: 'AppHeader',
  components: {
    ...VToolbar,
    ...VList,
    ...VNavigationDrawer
  },
  data () {
    return {
      routerSub: this.$router.history.current.matched[0].path,
      drawer: true,
      mini: true
    };
  },
  watch: {
    '$route' (to) {
      this.routerSub = to.matched[0].path;
    }
  },
  created () {
    this.drawerCtrl();
  },
  methods: {
    drawerCtrl () {
      if (window.innerWidth < 1264) {
        this.mini = false;
        return this.drawer = !this.drawer;
      } else {
        return this.mini = !this.mini;
      }
    },
    logout () {
      auth.logout().then((data) => {
        this.$router.push('/login');
      }).catch((error) => {
        alert('로그아웃 에러');
        /*

         * */
      });
    }
  }
};
